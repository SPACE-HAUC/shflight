143 typedef enum
144 {
145  PING = 1, // 1 byte, 0 data in, 1 byte out, 0 data out
146  REBOOT = 4, // 1 byte, 0x80 0x07 0x80 0x07 in, 0 out (reboot!)
147  GET_HK = 8, // request empty, reply struct hkparam_t
148  // request 0x00, reply eps_hk_t
149  // request 0x01, reply eps_hk_vi_t
150  // request 0x02, reply eps_hk_out_t
151  // request 0x03, reply eps_hk_wdt_t
152  // request 0x04, reply eps_hk_basic_t
153  SET_OUTPUT, // request 0x[XX], reply none
154  // Set output switch states by a bitmask where 1 means channel is on, 0 means channel is off
155  // LSB is channel 1, next bit is channel 2 etc
156  // MSB -- NC NC 3V3_3 3V3_2 3V3_1 5V_3 5V_2 5V_1 --LSB
157  SET_SINGLE_OUTPUT, // request: byte channel, byte value, int16 delay. Reply: None
158  // set output %channel% to value %value% with delay,
159  // channel (0-5), BP4 heater (6), BP4 switch (7)
160  // Value 0 = off, 1 = on; delay in seconds
161  SET_PV_VOLT, // request: uint16 voltage1, voltage2, voltage3, reply: none
162  // set the voltage on the photovoltaic inputs V1, V2, V3 in mV
163  // takes effect in MODE=2 (see SET_PV_AUTO)
164  // transmit voltage1 first and voltage3 last.
165  SET_PV_AUTO, // request: uint8 MODE, reply: none
166  // Sets the solar cell power tracking mode.
167  // MODE = 0: Hardware default power point
168  // MODE = 1: Maximum power point tracking
169  // MODE = 2: Fixed software power point, value set with SET_PV_VOLT, default 4V
170  SET_HEATER, // request: uint8 cmd, heater, mode; reply: uint8 bp4_heater, uint8 onboard_heater
171  // cmd = 0: set heater on/off; Heater: 0 = BP4, 1 = Onboard, 2 = Both; Mode: 0 = OFF, 1 = ON
172  // Replies with heater modes. 0=Off, 1 = On. To do only query, send an empty message
173  RESET_COUNTERS = 15, // request: 0x42. reply: none
174  // Send this command to reset boot counters and WDT counters
175  RESET_WDT, // request: 0x78, reply: None.
176  // Use this command to reset (kick) the dedicated WDT.
177  CONFIG_CMD, // request: uint8 cmd; reply: none
178  // cmd = 1: restore default config
179  CONFIG_GET, // request: None, reply: struct eps_config_t
180  CONFIG_SET, // request: struct eps_config_t, reply: none
181  HARD_RESET, // request: none, reply: none
182  // Hard reset of the P31 cycles permament 5V, 3V3 and battery outputs.
