48 inline float q2isqrt(float x)
49 {
50  float xhalf = x * 0.5f; // calculate 1/2 x before bit-level changes
51  int i = *(int *)&x; // convert float to int for bit level operation
52  i = 0x5f375a86 - ((*(int *)&x) >> 1); // bit level manipulation to get initial guess (ref: http://www.lomont.org/papers/2003/InvSqrt.pdf)
53  x = *(float *)&i; // convert back to float
54  x = x * (1.5f - xhalf * x * x); // 1 round of Newton approximation
55  x = x * (1.5f - xhalf * x * x); // 2 round of Newton approximation
56  x = x * (1.5f - xhalf * x * x); // 3 round of Newton approximation
57  return x;
58 }
